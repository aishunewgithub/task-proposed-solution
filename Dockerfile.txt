# Use official slim Python image
FROM python:3.10-slim

ENV DEBIAN_FRONTEND=noninteractive

# Install system dependencies needed by pdf libraries and camelot/pdf2image
RUN apt-get update && apt-get install -y --no-install-recommends \
    build-essential \
    poppler-utils \   # for pdf rendering (pdfplumber)
    libgl1-mesa-glx \
    libglib2.0-0 \
    ghostscript \     # needed by camelot
    tesseract-ocr \
    tesseract-ocr-chi-sim \
    wget \
    ca-certificates \
    && rm -rf /var/lib/apt/lists/*

# Copy code
WORKDIR /app
COPY . /app

# Install pip deps
RUN pip install --upgrade pip
RUN pip install -r requirements.txt

# Expose nothing (script)
ENTRYPOINT ["python", "parse_pdfs.py"]
